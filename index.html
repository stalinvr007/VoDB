    Getting Started

    1. Create a class to represent a database table/model.
    2. Create properties to represents the fields for that model.
    3. Decorate the properties with [DbField, DbKey, DbIdentity, DbRequired, DbBind, DbIgnore] See Employee.cs
      • [DbField] -> Allows users to specify the physical name of the field;
      • [DbKey] -> Well... its a key...
      • [DbIdentity] -> The field is the Identity field increasing automatically;
      • [DbRequired] -> Allows the framework to check for the required fields before executing SQL commands.
      • [DbBind] -> Lets the framework know whats the key field in the associated table/entity. Use only when field names don't match.
      • [DbIgnore] -> Tells the framework to ignore a field from mapping.
    4. Create a ConnectionString entry in the app.config/web.config file with the name of your machine.
      • set the providerName to System.Data.SqlClient or other...

    Get one employee

    using (var session = new Session()) {
   Employee employee = session.GetById(Employee { EmployeeId = 1 });
}

    Get all employees

    using (var session = new Session()) {
   IEnumerable<Employee> employees = session.GetAll<Employee>();
}

    Filtering employees

    // simple query
using (var session = new Session()) {
   IEnumerable<Employee> employees = 
       session.GetAll<Employee>().Where(e => e.EmployeeId > 5);
}

// more complex query
using (var session = new Session()) {
   IEnumerable<Employee> employees = session.GetAll<Employee>()
       .Where(e => e.ReportsTo).In(
           session.GetAll<Employee>().Where(e => e.EmployeeId > 5)
       );
   // Or
   IEnumerable<Employee> employees = session.GetAll<Employee>().Where(e => e.ReportsTo.EmployeeId > 5)
}

    Sort employees

    using (var session = new Session()) {
   IEnumerable<Employee> employees = 
       session.GetAll<Employee>().OrderBy(e => e.City);
}
// or ...
using (var session = new Session()) {
   IEnumerable<Employee> employees = 
       session.GetAll<Employee>().OrderBy(e => e.City).Descending();
}

    Precompiled Queries

    var query = Select.All.From<Employee>().Where(e => e.ReportsTo).In(
    Select.All.From<Orders>()
);

IEnumerable<Employee> employees = session.ExecuteQuery<Employee>(query);

    Precompiled Queries Parameters

    var query = Select.All.From<Employee>()
    .Where(e => e.EmployeeId > Param.Get<int>())
    .and(e => e.EmployeeId < Param.Get<int>())
    .and(e => e.City).In(Param.Get<String>(), Param.Get<String>(), Param.Get<String>())
);

IEnumerable<Employee> employees = session.ExecuteQuery<Employee>(query, 1, 2, new[] { "Lisbon", "New York", "Sao Paulo" });

    Executing within a transaction

    // Executing operations within a transaction.
   using (var session = new Session()) {
      using (var transaction = session.BeginTransaction()) {

         // Make changes.

         // Rolls back any change to the database.
         transaction.RollBack();
         // Commits the changes within this transaction scope.
         transaction.Commit();

         using(var innerTrans = session.BeginTransaction()) {
             // Make changes. If the rollback occurs then all transactions are void.
         }
      }
   }

   // Or
   using (var session = new Session()) {
      session.WithinTransaction( () => {

          // Do something here

      }); // Commits the changes
   }

    Exception Handling

    VODB.Config.RegisterExceptionHandler(
    new OnPrimaryKeyViolationHandler()
);

class OnPrimaryKeyViolationHandler: IExceptionHandler
{
    public bool CanHandle(Exception exception)
    {
        return exception is SqlException &&
            exception.Message.Contains("Violation of PRIMARY KEY");
    }

    public void Handle(Exception exception)
    {
        // Do something here and rethrow the exception wrapped or not.
        throw new PrimaryKeyViolationException(exception);
    }
}

    Vodb ships with 3 native exception handlers (OnPrimaryKeyExceptionHandler, OnUniqueKeyExceptionHandler and OnTruncatedExceptionHandler).

    basically wraps the sql exceptions with Vodb exception types that derived from VodbException. This is used to unify the catch statement.
